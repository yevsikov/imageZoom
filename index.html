imageZoom js lib example

You just need to add `zoom-js` class for your images

* required: jQuery

Setup

<!-- Setup -->
<script src="//code.jquery.com/jquery-3.3.1.min.js"></script>
<script src="src/imageZoom.js"></script>


Thumbnails with link to big size image in data-attribute `data-fullsize`

<!-- Thumbnails with link to big size image in data-attribute `data-fullsize` -->
<img src="https://picsum.photos/260/180?image=1068"
   class="zoom-js"
   data-fullsize="https://picsum.photos/1024/768?image=1068"
/>

 

 

 



Thumbnails without link to big size image

<!-- Thumbnails without link to big size image -->
<img src="https://picsum.photos/600/400?image=1068"
    class="zoom-js"
    style="width:300px;height:200px"
/>

 



Custom styles

<!-- Custom styles -->
<style>
    #zoomedArea {
        border: 1px dashed #fff;
        border-radius: 25px;
        cursor: crosshair;
        z-index: 1000;
    }
</style>
<div id="zoomedArea" data-margin="30"></div>

 



Callback using

<!-- Callback using -->
<img src="https://picsum.photos/260/180?image=1068"
   class="zoom-js"
   data-fullsize="https://picsum.photos/1024/768?image=1068"
   data-callback="Action One"
/>

<script>
    $(document).ready(function () {
        $('#zoomedArea').click(function() {
            //TODO: start working with `data-callback` attribute
            alert( $(this).attr('data-callback') );
        });
    });
</script>

   



Wordpress example

<!-- Wordpress example -->
<img src="<?php echo get_the_post_thumbnail_url($post->ID,'thumbnail'); ?>"
